# kdt6 - 황지원
use sqlclass_db;


# Q1. 1960년에 노벨상 물리학상 쪼는 노벨 편화상을 수상한 사람의 정보를 출력
select year, category, fullname from nobel
where (year = 1960) and (category = 'Physics' or category = 'Peace');

# Q2. Albert Einstein이 수상한 연도와 수상분야(category), 출생대륙, 출생국가를 출력
select year, category, prize_amount, birth_continent, birth_country from nobel
where fullname = 'Albert Einstein';

# Q3. 1910년부터 2010년까지 노벨 평화상 수상자 명단 출력 (연도 오름차순)
select year, fullname, birth_country from nobel
where (category = 'Peace') and (year between 1910 and 2010);

# Q4. 전체 테이블에서 수장자 이름이 'John'으로 시작하는 수상자 모두 출력
select category, fullname from nobel
where (fullname like 'John%');

# Q5. 1964년 수상자 중에서 노벨화학상과 의학상을 제외한 수상자의 모든 정보를 수상자 이름을 기준으로 오름차순 정렬 후 출력
select * from nobel
where not (category = 'Chemistry' or category = 'Physiology or Medicine')
order by fullname asc;

# Q6. 2000년부터 2019년까지 노벨 문학상 수상자 명단 출력
select year, fullname, gender, birth_country from nobel
where (category = 'Literature') and (year between 2000 and 2019);

# Q7. 각 분야별 역대 수상자의 수를 내림차순으로 정렬후 출력 (group by, order by)
select category, prize_amount from nobel
group by category, prize_amount
order by prize_amount desc;

# Q8. 노벨 의학상이 있었던 연도를 모두 출력 (distinct 사용)
select year from nobel
where category = 'Physiology or Medicine';

# Q9. 노벨 의학상이 없었던 연도의 총 횟수를 출력
select count(*) from nobel
group by year, category
having not category = 'Physiology or Medicine';

# Q10. 여성 노벨 수상자 명단 출력
select fullname, category, birth_country from nobel
where gender = 'female';

# Q11. 수상자들의 출생 국가별 횟수 출력
select birth_country, count(*) from nobel
group by birth_country;

# Q12. 수상자의 출생국가가 'Korea'인 정보 모두 출력
select * from nobel
where birth_country = 'Korea';

# Q13. 수상자의 출신 국가가 ('Europe', 'North America', '')이 아닌 모든 정보 출력
select * from nobel
where not (birth_country = 'Europe' or birth_country = 'North America' or birth_country = '');

# Q14. 수상자의 출신 국가별로 그룹화 해서 수상 횟수가 10회 이상인 국가의 모든 정보는 출력하고
# 	국가별 수상횟수의 역순으로 출력
select birth_country, count(*) from nobel
group by birth_country
having count(*) >= 10
order by count(*) desc;

# Q15. 2회 이상 수상자 중에서 fullname이 공백이 아닌 경우를 출력하느네, fullname의 오름차순으로 출력
select fullname, count(*) from nobel
where not (fullname = '')
group by fullname
having count(*) >= 2
order by fullname asc;
